class Solution {
    public ListNode reverseList(ListNode head) {
        ListNode pre =null;
        ListNode cur= head;

        while(cur!=null){
            ListNode temp=cur.next;
            cur.next=pre;
            pre=cur;
            cur=temp;
        }
        return pre;
    }
}

class Solution {
    public ListNode reverseList(ListNode head) {
        while(head==null || head.next==null){
            return head;
        }

        ListNode res=reverseList(head.next);
        head.next.next=head;
        head.next=null;
        return res;
    }
}
